//JOBNAME  JOB  CLASS=A,
//            MSGCLASS=H,
//            NOTIFY=&SYSUID
//*
//* Generate PACK and SPACK tersed data.
//* Note that rerunning this job will generate datasets
//* that do not match the existing git versions because
//* there is metadata in the terse output.
//* The results from the decompress operation should
//* be valid.
//*
//* The prefix for the Zigi repository z/OS datasets:
//*
//         SET  PREFIX='USERID.TDTDATA'
//*
//TERSE    PROC DS=
//DELETE   EXEC PGM=IEFBR14
//PACK     DD   DISP=(MOD,DELETE),UNIT=SYSDA,SPACE=(TRK,1),
//            DSN=&DS..PACK
//SPACK    DD   DISP=(MOD,DELETE),UNIT=SYSDA,SPACE=(TRK,1),
//            DSN=&DS..SPACK
//*
//PACK     EXEC PGM=AMATERSE,PARM=PACK
//SYSPRINT DD   SYSOUT=*
//SYSUT1   DD   DISP=SHR,DSN=&DS
//SYSUT2   DD   DISP=(NEW,CATLG),LRECL=1024,
//            DSN=&DS..PACK,
//            SPACE=(TRK,(100,100),RLSE),
//            UNIT=SYSDA
//*
//PACK     EXEC PGM=AMATERSE,PARM=SPACK
//SYSPRINT DD   SYSOUT=*
//SYSUT1   DD   DISP=SHR,DSN=&DS
//SYSUT2   DD   DISP=(NEW,CATLG),LRECL=1024,
//            DSN=&DS..SPACK,
//            SPACE=(TRK,(100,100),RLSE),
